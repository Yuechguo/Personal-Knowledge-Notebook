### modify podman default storage path
1.stop all running container and remove all exited container：
``` shell
podman stop $(podman ps -q)  

podman rm $(podman ps -q) 
```

2.modify podman config file:
``` shell
vim /etc/container/storage.conf

runroot = "{path/to/dir}"
graphroot = "{path/to/dir}"
tmpdir = "{path/to/dir}"
imageCopyTmpDir = "{{path/to/dir}}"
```

3.move old image storage  to new dir:
```
mv {/old_path/to/dir} {/new_path/to/dir}

example:
mv /home/root124/IMG /mnt/docker-data/
``` 

4.rm old bolt_state.db in order to active config setting of storage.conf(if old bolt_state not be deleted)


### dockerfile for vllm && triton build
image build process：
``` shell
base_image -> install FA-> -> install old triton -> install vllm -> pip uninstall pytorch-triton-rocm -> install triton -> test server mode
```

detail usage:
``` shell
vllm commit :a6414b8
vllm patch commit:  ec5e3c5d38b6dca86bd156c27c5c80ea8b65bb97
triton commit : 2db56689b0d1268f09dd99cabe4ca940d710da7e
```

dockerfile:
``` shell
FROM e2e_image_731_fa:latest

WORKDIR /rocm

COPY rocm_triton.tar /rocm/

RUN tar -xvf rocm_triton.tar && python3 -m pip install triton-3.0.0-cp39-cp39-linux_x86_64.whl

RUN python3 -m pip install --upgrade numba \
    && git clone https://github.com/ROCm/vllm.git \
	&& cd vllm \
	&& git checkout a6414b8 \
	&& wget https://github.com/ROCm/vllm/commit/ec5e3c5d38b6dca86bd156c27c5c80ea8b65bb97.patch \
	&& git apply ec5e3c5d38b6dca86bd156c27c5c80ea8b65bb97.patch \
    && pip install -U -r requirements-rocm.txt \
    && pip install numpy==1.22.4 \
    && case "$(ls /opt | grep -Po 'rocm-[0-9]\.[0-9]')" in \
        *"rocm-6.0"*) \
            patch /opt/rocm/include/hip/amd_detail/amd_hip_bf16.h rocm_patch/rocm_bf16.patch;; \
        *"rocm-6.1"*) \
            cp rocm_patch/libamdhip64.so.6 /opt/rocm/lib/libamdhip64.so.6;; \
        *) ;; esac \
    && python3 setup.py install \
    && cd gradlib \
    && python3 setup.py install

RUN pip uninstall -y pytorch-triton-rocm \
	&& git clone https://github.com/triton-lang/triton.git \
	&& cd triton && git checkout 2db566 \
	&& cd python && pip install ninja cmake wheel && pip install -e .

RUN cp -r /rocm/vllm/vllm/entrypoints/sync_openai /opt/conda/envs/py_3.9/lib/python3.9/site-packages/vllm-0.4.3+rocm614-py3.9-linux-x86_64.egg/vllm/entrypoints/
```
