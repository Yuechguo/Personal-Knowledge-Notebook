### ubuntu update to rocm7.2.2
``` shell
# 删除 amdgpu安装的包，如果有的话，
$ apt remove amdgpu-install-internal
$ amdgpu-uninstall
#删除 repo list
$ rm /etc/apt/sources.list.d 裡面 amd, rocm 相關的 list
# update repo
$ apt update
# get 最新的repo list 
$ wget https://repo.radeon.com/amdgpu-install/7.2.2/ubuntu/jammy/amdgpu-install_7.2.2.70202-1_all.deb
# 安装最新repo list
$ apt install ./amdgpu-install_7.2.2.70202-1_all.deb
# 调整repo list的优先级
$ sudo tee /etc/apt/preferences.d/99-radeon-repo-priority >/dev/null <<'EOF'  
Package: *  
Pin: origin repo.radeon.com  
Pin-Priority: 1001  
EOF
# 删除 repo list下 graphic相关的东西
$ cd /etc/apt/sources.list.d/
$ grep -r "graphic"
rocm.list:deb [arch=amd64,i386 signed-by=/etc/apt/keyrings/rocm.gpg] https://repo.radeon.com/graphics/7.2.2/ubuntu jammy main
删除掉类似的行
# 安装package
amdgpu-install --usecase=rocm --no-dkms -y
```
